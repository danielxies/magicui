## Usage

```tsx
import { ProgressiveBlur } from "@/components/ui/progressive-blur";

export default function MyComponent() {
  return (
    <div className="relative h-[400px] overflow-auto">
      {/* Your scrollable content */}
      <div className="space-y-4">{/* Items, text, etc. */}</div>

      <ProgressiveBlur />
    </div>
  );
}
```

## Props

| Prop         | Type                          | Default                         | Description                                     |
| ------------ | ----------------------------- | ------------------------------- | ----------------------------------------------- |
| `className`  | `string`                      | `undefined`                     | Optional CSS class to apply to the component    |
| `height`     | `string`                      | `"65%"`                         | Height of the blur gradient                     |
| `position`   | `"top" \| "bottom" \| "both"` | `"bottom"`                      | Position of the blur effect                     |
| `blurLevels` | `number[]`                    | `[0.5, 1, 2, 4, 8, 16, 32, 64]` | Blur levels for each gradient layer             |
| `children`   | `React.ReactNode`             | `undefined`                     | Optional content to render inside the component |
