## Usage

```tsx
import VideoText from "@/components/magicui/video-text";

export default function VideoTextExample() {
  return (
    <div className="h-[400px] w-full bg-black">
      <VideoText
        text="your-text-here"
        sources={[
          {
            src: "path/to/video.webm",
            type: "video/webm",
          },
          {
            src: "path/to/video.mp4",
            type: "video/mp4",
          },
        ]}
      />
    </div>
  );
}
```

## Props

| Prop        | Type                             | Default  | Description                                                                                                                    |
| ----------- | -------------------------------- | -------- | ------------------------------------------------------------------------------------------------------------------------------ |
| `text`      | `string`                         | Required | The text to display over the video                                                                                             |
| `sources`   | `{ src: string, type: string}[]` | Required | Array of video sources with type. Multiple sources improve browser compatibility. The `src` can be a URL or a local file path. |
| `autoPlay`  | `boolean`                        | `true`   | Whether to autoplay the video                                                                                                  |
| `muted`     | `boolean`                        | `true`   | Whether to mute the video                                                                                                      |
| `loop`      | `boolean`                        | `true`   | Whether to loop the video                                                                                                      |
| `preload`   | `'auto' \| 'metadata' \| 'none'` | `"auto"` | Video preload behavior                                                                                                         |
| `className` | `string`                         | `""`     | Additional className for the container                                                                                         |

## Types

```ts
interface VideoSource {
  src: string; // URL or local file path
  type: string; // MIME type like "video/webm" or "video/mp4"
}
```
